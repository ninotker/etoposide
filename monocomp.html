def find_func(temps, f_lib, name):
    def monocomp_function(t, A, B, k):
        return A + B * np.exp(-k * t)
    popt, pcov = curve_fit(monocomp_function, temps, f_lib, p0=[max(f_lib), 1, 0.1], maxfev=10000)
    A, B, k = popt
    return round(A, 2), round(B, 2), round(k, 2)
